public class PojistenciModel : PageModel
{
    private readonly ApplicationDbContext DB;

    public List<Pojistenec> Data { get; set; } = new List<Pojistenec>();
    public int CurrentPage { get; set; } = 1;
    public int TotalPages { get; set; }
    private const int PageSize = 10; // Počet pojištěnců na stránku

    public PojistenciModel(ApplicationDbContext db)
    {
        DB = db;
    }

    public async Task OnGetAsync(int? pageNumber)
    {
        // Nastavení aktuální stránky
        CurrentPage = pageNumber ?? 1;

        // Počet všech pojištěnců v databázi
        int totalRecords = await DB.Pojistenci.CountAsync();

        // Výpočet celkového počtu stránek
        TotalPages = (int)Math.Ceiling(totalRecords / (double)PageSize);

        // Načtení pojištěnců pro aktuální stránku
        Data = await DB.Pojistenci
            .OrderBy(x => x.Prijmeni)
            .Skip((CurrentPage - 1) * PageSize)
            .Take(PageSize)
            .ToListAsync();
    }
}


@page
@model PojistenciModel
@{
    ViewData["Title"] = "Seznam pojištěnců";
}

<h1>Seznam pojištěnců</h1>

<table class="table table-striped">
    <thead>
        <tr>
            <th>ID</th>
            <th>Jméno</th>
            <th>Příjmení</th>
            <th>Datum narození</th>
            <th>Číslo pojištění</th>
        </tr>
    </thead>
    <tbody>
        @foreach (var pojistenec in Model.Data)
        {
            <tr>
                <td>@pojistenec.Id</td>
                <td>@pojistenec.Jmeno</td>
                <td>@pojistenec.Prijmeni</td>
                <td>@pojistenec.DatumNarozeni.ToString("d.M.yyyy")</td>
                <td>@pojistenec.CisloPojisteni</td>
            </tr>
        }
    </tbody>
</table>

<nav aria-label="Stránkování">
    <ul class="pagination justify-content-center">
        <!-- Tlačítko Předchozí -->
        <li class="page-item @(Model.CurrentPage <= 1 ? "disabled" : "")">
            <a class="page-link" asp-page="./Pojistenci" asp-route-pageNumber="@(Model.CurrentPage - 1)">Předchozí</a>
        </li>

        <!-- Číslované stránky -->
        @for (int i = 1; i <= Model.TotalPages; i++)
        {
            <li class="page-item @(i == Model.CurrentPage ? "active" : "")">
                <a class="page-link" asp-page="./Pojistenci" asp-route-pageNumber="@i">@i</a>
            </li>
        }

        <!-- Tlačítko Další -->
        <li class="page-item @(Model.CurrentPage >= Model.TotalPages ? "disabled" : "")">
            <a class="page-link" asp-page="./Pojistenci" asp-route-pageNumber="@(Model.CurrentPage + 1)">Další</a>
        </li>
    </ul>
</nav>
